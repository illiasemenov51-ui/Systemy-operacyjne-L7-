#!/bin/bash

while true; do
  echo "Podaj numer albumu:"
  read numer

  ./numer_indeksu "$numer"
  wynik=$?   # kod błędu z poprzedniego programu

  if [ $wynik -eq 0 ]; then
    echo "✅ Poprawny numer albumu!"
    break
  else
    echo "❌ Błędny numer albumu, spróbuj ponownie."
  fi
done
